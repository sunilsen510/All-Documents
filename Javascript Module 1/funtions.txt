// 1Ô∏è‚É£ Array to store tasks
let tasks = [];

// 2Ô∏è‚É£ DOM Elements
const input = document.getElementById("todo-input");
const addBtn = document.getElementById("add-btn");
const list = document.getElementById("todo-list");

// 3Ô∏è‚É£ Function to display tasks
function displayTasks() {
  list.innerHTML = ""; // Clear existing list
  tasks.forEach((task, index) => {
    const li = document.createElement("li");
    li.textContent = task.text;
    if (task.completed) li.classList.add("completed");

    // Toggle complete on click
    li.addEventListener("click", () => {
      toggleTask(index);
    });

    // Delete button
    const delBtn = document.createElement("button");
    delBtn.textContent = "Delete";
    delBtn.className = "delete";
    delBtn.addEventListener("click", (e) => {
      e.stopPropagation(); // Prevent li click
      deleteTask(index);
    });

    li.appendChild(delBtn);
    list.appendChild(li);
  });
}

// 4Ô∏è‚É£ Function to add task
function addTask(taskText, callback) {
  if (taskText.trim() === "") {
    alert("Please enter a task!");
    return;
  }
  tasks.push({ text: taskText, completed: false });
  callback(); // Call display function after adding
}

// 5Ô∏è‚É£ Function to delete task
function deleteTask(index) {
  tasks.splice(index, 1);
  displayTasks();
}

// 6Ô∏è‚É£ Function to toggle complete
function toggleTask(index) {
  tasks[index].completed = !tasks[index].completed;
  displayTasks();
}

// 7Ô∏è‚É£ Event Listener for Add Button
addBtn.addEventListener("click", () => {
  addTask(input.value, () => {
    displayTasks();
    input.value = ""; // Clear input
  });
});

// Optional: Enter key press
input.addEventListener("keypress", (e) => {
  if (e.key === "Enter") addBtn.click();
});

// Initial Display
displayTasks();
